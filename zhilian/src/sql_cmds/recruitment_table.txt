CREATE TABLE IF NOT EXISTS `Recruitment`(
	`Job_ID` varchar(200) NOT NULL,
	`Company_ID` varchar(200) NOT NULL,

	`jobName` varchar(200) NULL DEFAULT NULL,
    `jobType` varchar(500) NULL DEFAULT NULL,
    `jobTagZhiLian` varchar(500) NULL DEFAULT NULL,
    `detailAddr` varchar(500) NULL DEFAULT NULL,
    `description` varchar(2000) NULL DEFAULT NULL,

	`salary` varchar(200) NULL DEFAULT NULL,
	`welfare` varchar(500) NULL DEFAULT NULL,

	`workingExp` varchar(200) NULL DEFAULT NULL,
	`eduLevel` varchar(200) NULL DEFAULT NULL,

	`resumeCount` varchar(100) NULL DEFAULT NULL,
	`recruitCount` varchar(100) NULL DEFAULT NULL,

	`positionURL` varchar(200) NULL DEFAULT NULL,
	`emplType` varchar(100) NULL DEFAULT NULL,

	`recentAndTotal` varchar(500) NULL DEFAULT NULL,
	`feedbackRation` FLOAT(6,5) NULL DEFAULT NULL,
	`interview` INT(20) NULL DEFAULT NULL,
	`selected` INT(20) NULL DEFAULT NULL,
	`applied` INT(20) NULL DEFAULT NULL,
	`collected` INT(20) NULL DEFAULT NULL,

	`jobTagManual` varchar(2000) NULL DEFAULT NULL,

	`Source_Website` varchar(200) NULL DEFAULT NULL,
	`Update_Name` varchar(100) NULL DEFAULT NULL,
	`Update_Time` DATETIME NULL DEFAULT NULL,

	 PRIMARY KEY(`Job_ID`)
)
COLLATE='utf8_general_ci'
ENGINE=MyISAM
;

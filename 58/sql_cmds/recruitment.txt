CREATE TABLE IF NOT EXISTS `recruitment` (
  `Job_ID` varchar(200) NOT NULL,
  `jobTitle` varchar(100) DEFAULT NULL,
  `jobSubTitle` varchar(200) DEFAULT NULL,
  `jobSalary` varchar(100) DEFAULT NULL,
  `applyNum` int(10) DEFAULT NULL,
  `resumeReadPercent` int(10) DEFAULT NULL,
  `browserNum` varchar(100) DEFAULT NULL,
  `jobWelfare` varchar(200) DEFAULT NULL,
  `jobRequirement` varchar(100) DEFAULT NULL,
  `jobAddr` varchar(200) DEFAULT NULL,
  `jobDescription` varchar(2000) DEFAULT NULL,
  `jobCity` varchar(10) DEFAULT NULL,
  `Update_Name` varchar(100) DEFAULT NULL,
  `Update_Time` datetime DEFAULT NULL,
  PRIMARY KEY (`Job_ID`)
) ENGINE=InnoDB COLLATE=utf8mb4_general_ci;
